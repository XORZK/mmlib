451template <typename T>
452vector<T> vector<T>::cross(const vector<T> v2) const {
453 assert(this->size == 3 && v2.get_size() == 3);
454 vector<T> i = vector<T>::i(), j = vector<T>::j(), k = vector<T>::k();
455
456 return (i * (this->data[1]*v2[2] - this->data[2]*v2[1])) +
457 (j * (this->data[2]*v2[0] - this->data[0]*v2[2])) +
458 (k * (this->data[0]*v2[1] - this->data[1]*v2[0]));
459}
