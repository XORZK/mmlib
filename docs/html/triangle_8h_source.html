50inline vector<vector<double>> sort_vertices(const vector<vector<double>>& vertices) {
51 vector<vector<double>> sorted(vertices.get_size());
52 vector<double> C = (vertices[0].get_size() == 3 ? create_vec3(0,0,1) : create_vec2(0, 0));
53
54 for (uint64_t i = 0; i < vertices.get_size(); i++) {
55 C[0] += vertices[i][0];
56 C[1] += vertices[i][1];
57 }
58
59 C /= vertices.get_size();
60
61 for (uint64_t i = 0; i < vertices.get_size(); i++) {
62 vector<double> n = (vertices[i] - C);
63 double dist = distance(n[0], n[1], 0.0, 0.0);
64
65 double angle = atan2(n[1], n[0]) * 180/M_PI;
66 angle += (360 * (angle < 0));
67 sorted.push_back(create_vec3(angle, dist, static_cast<double>(i)));
68 }
69
70 // Sort the vector
71 // Insertion Sort? --> Merge Sort (TODO)
72 for (uint64_t i = 1; i < sorted.get_size(); i++) {
73 uint64_t k = i;
74 while (k > 0 && ((sorted[k][0] < sorted[k-1][0]) || (sorted[k][0] == sorted[k-1][0] && sorted[k][1] < sorted[k-1][1]))) {
75 sorted.swap(k, k-1);
76 k--;
77 }
78 }
79
80 for (uint64_t i = 0; i < sorted.get_size(); i++) {
81 sorted[i] = vertices[sorted[i][2]];
82 }
83
84 return sorted;
85}

88inline vector<double> centroid(vector<vector<double>> vertices) {
89 vertices = sort_vertices(vertices);
90 int64_t N = vertices.get_size();
91 // A is the polygon's signed area -> A = (1/2) sum_{i=0}{n-1} [(x_i * y_(i+1)) - (x_(i+1) * y_i)]
92 double A = 0.0, Cx = 0.0, Cy = 0.0;
93 for (uint64_t i = 0; i < N; i++) {
94 A += (vertices[i][0] * vertices[(i+1)%N][1]) - (vertices[(i+1)%N][0] * vertices[i][1]);
95 }
96
97 A /= 2.0;
98
99 assert(A != 0);
100
101 for (uint64_t i = 0; i < N; i++) {
102 double T = (vertices[i][0] * vertices[(i+1)%N][1] - vertices[(i+1)%N][0] * vertices[i][1]);
103
104 Cx += (vertices[i][0] + vertices[(i+1)%N][0]) * T;
105 Cy += (vertices[i][1] + vertices[(i+1)%N][1]) * T;
106 }
107
108 Cx /= (6*A);
109 Cy /= (6*A);
110
111 return create_vec2(Cx, Cy);
112}

114// Computes the Barycentric Coordinates of a Point (possibly) within a Triangle ABC
115// alpha = A(PCB)/A
116// beta = A(PCA)/A
117// gamma = 1 - (alpha + beta)
118// We use the cross product to determine the area of triangles ABC, APC, APB, BPC
119// which are then used to determine the weights (alpha, beta, gamma).
120inline vector<double> barycentric(vector<double> p,
121 vector<double> a,
122 vector<double> b,
123 vector<double> c) {
124
125 if (p.get_size() == 2) { p = create_vec3(p); }
126 if (a.get_size() == 2) { a = create_vec3(a); }
127 if (b.get_size() == 2) { b = create_vec3(b); }
128 if (c.get_size() == 2) { c = create_vec3(c); }
129
130 double A = ((b-a).cross(c-a).magnitude()),
131 AB = ((a-p).cross(b-p).magnitude()),
132 AC = ((a-p).cross(c-p).magnitude()),
133 BC = ((b-p).cross(c-p).magnitude());
134
135 double alpha = BC/A, beta = AC/A, gamma = 1 - (alpha + beta);
136
137 return create_vec3(alpha, beta, gamma);
138}

186inline vector<triangle> triangulate(vector<vector<double>>& vertices, vector<int64_t>& reflex, vector<int64_t>& ears) {
187 vector<triangle> out;
188 // If the set of vertices does not form a triangle, remove an ear vertex
189 // and update the status of it's adjacent vertices
190 // { P1, P2, P3, P4, P5 } -> { P1, P2, P4, P5 }
191 // We want to update using Angles P1-P2-P4 and P2-P4-P5
192 // If we remove index N, we update vertexes (N-1, N, N+1) and (N, N+1, N+2)
193 while (vertices.get_size() > 3) {
194 int64_t N = vertices.get_size();
195 int64_t removal = ears[0];
196
197 triangle t = triangle(vertices[mod(removal-1, N)],
198 vertices[mod(removal, N)],
199 vertices[mod(removal+1,N)]);
200
201 ears.remove(0);
202 vertices.remove(removal);
203
204 for (int64_t i = 0; i < ears.get_size(); i++) {
205 if (ears[i] > removal) {
206 ears[i]--;
207 }
208 }
209
210 for (int64_t i = 0; i < reflex.get_size(); i++) {
211 if (reflex[i] > removal) {
212 reflex[i]--;
213 }
214 }
215
216 update_polygon_vertex(vertices, reflex, ears, removal);
217
218 out.push_back(t);
219 }
220
221 out.push_back(triangle(vertices[0], vertices[1], vertices[2]));
222
223 return out;
224}

227inline vector<triangle> triangulate(vector<vector<double>> vertices) {
228 vertices = sort_vertices(vertices);
229 vector<int64_t> reflex, ears;
230 int64_t N = vertices.get_size();
231
232 // By cosine law: c^2 = a^2 + b^2 - 2*a*b*cos(C)
233 // cos(C) = (a^2 + b^2 - c^2)/2*a*b
234 for (int64_t i = 0; i < N; i++) {
235 vector<double> vA = vertices[mod(i-1, N)], vB = vertices[mod(i, N)], vC = vertices[mod(i+1, N)];
236
237 // If det(b-a, c-b) > 0 --> vertices form a left turn -> implies convex
238 // (b.x - a.x) * (c.y - b.y) - (c.x - b.x) * (b.y - a.y)
239 if (!is_convex_vertex(vertices, i)) {
240 reflex.push_back(i);
241 } else {
242 if (is_ear_vertex(vertices, i)) {
243 ears.push_back(i);
244 }
245 }
246 }
247
248 return triangulate(vertices, reflex, ears);
249}
